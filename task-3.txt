<script>

const quiz=[

{
question:"Which language is used to style webpages?",
answers:[
{text:"HTML",correct:false},
{text:"CSS",correct:true},
{text:"Python",correct:false},
{text:"Java",correct:false}
]
},

{
question:"Which language adds interactivity?",
answers:[
{text:"JavaScript",correct:true},
{text:"CSS",correct:false},
{text:"HTML",correct:false},
{text:"C",correct:false}
]
},

{
question:"HTML stands for?",
answers:[
{text:"Hyper Text Markup Language",correct:true},
{text:"High Text Machine Language",correct:false},
{text:"Home Tool Markup Language",correct:false},
{text:"None",correct:false}
]
},

{
question:"Which symbol is used for comments in JavaScript?",
answers:[
{text:"//",correct:true},
{text:"<!-- -->",correct:false},
{text:"#",correct:false},
{text:"**",correct:false}
]
},

{
question:"Which company developed JavaScript?",
answers:[
{text:"Netscape",correct:true},
{text:"Microsoft",correct:false},
{text:"Google",correct:false},
{text:"Apple",correct:false}
]
}

];

let current=0;
let marks=0;

const question=document.getElementById("question");
const answers=document.getElementById("answers");
const nextBtn=document.getElementById("nextBtn");
const score=document.getElementById("score");

function loadQuestion(){

answers.innerHTML="";
nextBtn.style.display="none";

question.innerHTML=(current+1)+". "+quiz[current].question;

quiz[current].answers.forEach(answer=>{

let btn=document.createElement("button");

btn.innerHTML=answer.text;

btn.classList.add("btn");

btn.onclick=()=>selectAnswer(btn,answer.correct);

answers.appendChild(btn);

});

}

function selectAnswer(button,correct){

Array.from(answers.children).forEach(btn=>btn.disabled=true);

if(correct){
button.classList.add("correct");
marks++;
}else{
button.classList.add("wrong");
}

Array.from(answers.children).forEach((btn,index)=>{
if(quiz[current].answers[index].correct){
btn.classList.add("correct");
}
});

nextBtn.style.display="block";
}

nextBtn.onclick=function(){

current++;

if(current<quiz.length){
loadQuestion();
}else{
showResult();
}

}

function showResult(){

question.innerHTML="🎉 Quiz Completed!";

answers.innerHTML="";

score.innerHTML="Your Score : "+marks+" / "+quiz.length;

nextBtn.innerHTML="Restart";

nextBtn.style.display="block";

nextBtn.onclick=function(){

current=0;
marks=0;
score.innerHTML="";
nextBtn.innerHTML="Next";
loadQuestion();

}

}

loadQuestion();

</script>
